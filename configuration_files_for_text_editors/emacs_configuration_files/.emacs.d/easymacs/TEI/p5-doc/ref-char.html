TEI P5: Guidelines for Electronic Text Encoding and Interchange

<char>

char(character) provides descriptive information about a character.
Attributes: Global attributes only
Declaration
element char
{
   att.global.attributes,
   (
      charName,
      macro.glossSeq,
      charProp*,
      mapping*,
      model.graphicLike*,
      model.noteLike?
   )
}
Example
 <char xml:id="circledU4EBA">
  <charName>CIRCLED IDEOGRAPH 4EBA</charName>
  <charProp>
   <unicodeName>character-decomposition-mapping</unicodeName>
   <value>circle</value>
  </charProp>
  <charProp>
   <localName>daikanwa</localName>
   <value>36</value>
  </charProp>
  <mapping type="standard">
   <g ref="#U4EBA">人</g>
  </mapping>
 </char>
